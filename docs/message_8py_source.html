150 def bytes_to_message(binary):
151  if not binary:
152  return None
153  _msg_length = struct.unpack_from('!i', binary, 0)
154  # 因为会返回元组，只能够这样加上索引来访问,下面同理
155  msg_length = _msg_length[0]
156  if msg_length == 0:
157  return KeepAlive()
158  else:
159  _msg_id = struct.unpack_from('!b', binary, 4)
160  msg_id = _msg_id[0]
161  if msg_id == 0:
162  return Choke()
163  elif msg_id == 1:
164  return UnChoke()
165  elif msg_id == 2:
166  return Interested()
167  elif msg_id == 3:
168  return UnInterested()
169  elif msg_id == 4:
170  _piece_index = struct.unpack_from('!i', binary, 5)
171  return Have(_piece_index[0])
172  elif msg_id == 5:
173  tuple_bitfield = struct.unpack_from('!'+str(msg_length-1)+'b', binary, 5)
174  print(tuple_bitfield)
175  list_bitfield = list(tuple_bitfield)
176  return Bitfield(list_bitfield)
177  elif msg_id == 6:
178  _piece_index = struct.unpack_from('!i', binary, 5)
179  return Request(_piece_index[0])
180  elif msg_id == 7:
181  _piece_index = struct.unpack_from('!i', binary, 5)
182  piece_index = _piece_index[0]
183  raw_data = binary[9:]
184  return Piece(piece_index, raw_data)
185  return None
